	mul // a * c
	mul // 4 * (a * c)
	sub // (b^2) - (4 * (a * c)) 

	popr ax // записываем вершину стека в ax

	jae return

	// ветвление ради ветвления

	popr ax

return:
	ret // откатывает стек в состояние до вызова
